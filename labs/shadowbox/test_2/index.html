Demonstrates a workaround for asynchronous loading of Shadowbox (with LABjs).

Shadowbox.init() uses document.write() to include dependencies — which won’t work if shadowbox.js is loaded after page load — so instead we load the required files explictly within the $LAB.script() call:

$LAB
.script('../shadowbox/shadowbox.js')
.block(function(){
	$LAB
	.script('../shadowbox/languages/shadowbox-en.js')
	.script('../shadowbox/players/shadowbox-iframe.js')
	.block(function(){
		$LAB
		.script('../shadowbox/adapters/shadowbox-jquery.js')
	});
});
